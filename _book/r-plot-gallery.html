getrank <- function(data)
{
  group <- nrow(data)/4
  rankvalue <- NULL
  for(i in 1:group)
  {
    rankvalue <- c(rankvalue, rank(ifelse(data$stat[i*4] %in% c("n_over", "risk"), 1, -1) *      
                                     data$value[((i-1)*4+1):(i*4)], ties.method = "min"))
  }
  data$rank <- as.character(rankvalue)
  return(data)
}
